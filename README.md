## Install

```bash
npm install --save cirrent-cdn
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'cirrent-cdn'
import 'cirrent-cdn/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
